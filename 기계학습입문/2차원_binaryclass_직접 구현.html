def logistic2(x0,x1,w):  #시그모이드 함수
    y=1/(1+np.exp(-(w[0]*x0+w[1]*x1+w[2])))
    return y

def cee_logistic2(w,x,t):
    X_n=x.shape[0]
    y=logistic2(x[:,0],x[:,1],w)
    cee=0
    for n in range(len(y)):
        cee= cee-(t[n,0]*np.log(y[n])+(1-t[n,0])*np.log(1-y[n])) #cross entropy
    
    cee=cee/X_n
    return cee
